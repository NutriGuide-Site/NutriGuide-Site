    <script>
        let currentIndex = 0;
        const formContainers = document.querySelectorAll('.form-container');
        const nextBtns = document.querySelectorAll('button[id^="nextBtn"]');
        const backBtns = document.querySelectorAll('button[id^="backBtn"]');
        const submitBtn = document.getElementById('submitBtn');

        function showQuestion(index) {
            formContainers.forEach((form, i) => {
                form.style.display = (i === index) ? 'block' : 'none';
            });
        }

        nextBtns.forEach((btn, index) => {
            btn.addEventListener('click', () => {
                if (index < formContainers.length - 1) {
                    currentIndex++;
                    showQuestion(currentIndex);
                }
            });
        });

        backBtns.forEach((btn, index) => {
            btn.addEventListener('click', () => {
                if (index > 0) {
                    currentIndex--;
                    showQuestion(currentIndex);
                }
            });
        });

        submitBtn.addEventListener('click', () => {
            alert('Formulaire soumis!');
        });

        // Toggle class 'selected' when clicking on options
        const options = document.querySelectorAll('.option');
        options.forEach(option => {
            option.addEventListener('click', () => {
                options.forEach(o => o.classList.remove('selected'));
                option.classList.add('selected');
            });
        });

        showQuestion(currentIndex);
    </script>
